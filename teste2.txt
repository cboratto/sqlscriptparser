DECLARE
  RESOURCE_BUSY EXCEPTION;
  PRAGMA EXCEPTION_INIT(RESOURCE_BUSY, -00054);
  NAOEXECUTOU BOOLEAN := TRUE;
BEGIN
  WHILE NAOEXECUTOU
  LOOP
    BEGIN
      EXECUTE IMMEDIATE 'ALTER TABLE SCHEMA_OWNER.PERSON_EXTERNAL_DATA_CBO ADD CONSTRAINT PERSEXTEDAT_PERSEXTEDATACBO_FK FOREIGN KEY(IDT_PERSON_EXTERNAL_DATA) REFERENCES SCHEMA_OWNER.PERSON_EXTERNAL_DATA(IDT_PERSON_EXTERNAL_DATA) ENABLE NOVALIDATE';
      NAOEXECUTOU := FALSE;
    EXCEPTION
      WHEN RESOURCE_BUSY THEN 
        DBMS_LOCK.SLEEP(SECONDS =>1);
      WHEN OTHERS THEN
        RAISE;
    END;
  END LOOP;
END;
